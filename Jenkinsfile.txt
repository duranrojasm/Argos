pipeline {
    agent any
    stages {
        stage('Checkout-git') {
            steps {
                git poll: true, url: 'https://github.com/duranrojasm/Argos.git'
            }
        }
        stage('Build') {
            steps {
                bat 'echo "Hola Mundo"'
            }
        }
		stage('Check'){
	    	steps {
                script{
   		    		parentBranches = '$(git branch)'
   		    		bat 'echo ${parentBranches}'
				}
            }
		}
    }
}
